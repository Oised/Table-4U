CREATE OR REPLACE FUNCTION log_delete_funcionario()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO log_funcionario (funcionario_id, nome, acao)
  VALUES (OLD.funcionario_id, OLD.nome, 'DELETE');

  RETURN OLD;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_delete_funcionario
AFTER DELETE ON funcionario
FOR EACH ROW
EXECUTE FUNCTION log_delete_funcionario();
